let currentQ = 0;
let answers = new Array(questions.length).fill(null);

function startQuiz() {
  document.getElementById('screen-start').classList.add('hidden');
  document.getElementById('screen-quiz').classList.remove('hidden');
  renderQuestion();
}

function renderQuestion() {
  const q = questions[currentQ];
  const container = document.getElementById('question-container');
  const selectedVal = answers[currentQ];

  const optionsHtml = q.options.map((opt, i) => `
    <button class="option ${selectedVal === i ? 'selected' : ''}" data-index="${i}">
      <span class="option-dot"></span>
      <span class="option-text">${opt.text}</span>
    </button>
  `).join('');

  container.innerHTML = `
    <div class="question-card">
      <div class="question-number">Question ${currentQ + 1} sur ${questions.length}</div>
      <div class="question-text">${q.text}</div>
      <div class="options">${optionsHtml}</div>
    </div>
  `;

  document.getElementById('progress-count').textContent = `${currentQ + 1} / ${questions.length}`;
  document.getElementById('progress-fill').style.width = `${((currentQ + 1) / questions.length) * 100}%`;
  document.getElementById('btn-back').style.visibility = currentQ === 0 ? 'hidden' : 'visible';
  document.getElementById('btn-next').disabled = selectedVal === null;
  document.getElementById('btn-next').textContent = currentQ === questions.length - 1 ? 'Voir mes résultats →' : 'Continuer →';
}

function selectOption(i) {
  answers[currentQ] = i;
  renderQuestion();
}

function nextQuestion() {
  if (answers[currentQ] === null) return;
  if (currentQ < questions.length - 1) {
    currentQ++;
    renderQuestion();
  } else {
    document.getElementById('screen-quiz').classList.add('hidden');
    document.getElementById('screen-email').classList.remove('hidden');
  }
}

function prevQuestion() {
  if (currentQ > 0) {
    currentQ--;
    renderQuestion();
  }
}

function computeProfile() {
  const scores = [0, 0, 0];
  answers.forEach((answerIdx, qIdx) => {
    if (answerIdx !== null) {
      const s = questions[qIdx].options[answerIdx].score;
      scores[0] += s[0];
      scores[1] += s[1];
      scores[2] += s[2];
    }
  });
  const maxIdx = scores.indexOf(Math.max(...scores));
  return profiles[maxIdx];
}

async function sendToBrevo(name, email, profileId) {
  if (BREVO_API_KEY === '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') {
    console.warn('[Brevo] API key non configurée — contact non enregistré.');
    return;
  }

  const listId = BREVO_LIST_IDS[profileId];
  if (!listId) {
    console.warn('[Brevo] ID de liste non configuré pour le profil :', profileId);
    return;
  }

  try {
    // Créer ou mettre à jour le contact
    // Si le contact existe déjà, Brevo le met à jour sans erreur (upsert)
    const res = await fetch('https://api.brevo.com/v3/contacts', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'api-key': BREVO_API_KEY
      },
      body: JSON.stringify({
        email: email,
        // Attributs du contact (doivent exister dans Brevo → Contacts → Attributs)
        attributes: {
          PRENOM:            name !== 'toi' ? name : undefined,
          PROFIL_QUIZ:       profileId,               // ex : "eveil"
          RGPD_CONSENT:      true,
          RGPD_DATE:         new Date().toISOString(), // horodatage du consentement
          RGPD_SOURCE:       'quiz-shift-identitaire'
        },
        // Ajouter dans la liste correspondant au profil
        // → déclenche automatiquement l'automation Brevo associée
        listIds: [listId],
        // Double opt-in désactivé : le consentement est explicite via la case RGPD
        updateEnabled: true  // met à jour le contact s'il existe déjà
      })
    });

    if (res.status === 201) {
      console.log('[Brevo] ✓ Contact créé et ajouté à la liste', listId);
    } else if (res.status === 204) {
      console.log('[Brevo] ✓ Contact mis à jour, ajout à la liste', listId);
    } else {
      const err = await res.json();
      console.error('[Brevo] Erreur :', res.status, err);
    }

  } catch (e) {
    // Erreur réseau — on n'empêche pas l'affichage du résultat
    console.error('[Brevo] Erreur réseau :', e);
  }
}

<script>
// ── Master event dispatcher ───────────────────────────────────────────────────
// Single delegated listener on document — survives any innerHTML rewrite.
// Buttons are identified by id or class, not by onclick.
document.addEventListener('click', function(e) {
  var t = e.target;

  // Walk up max 3 levels to find a clickable element
  // (handles clicks on child spans inside buttons)
  for (var i = 0; i < 3; i++) {
    if (!t || t === document) break;

    // START button
    if (t.classList && t.classList.contains('btn-start')) {
      e.preventDefault();
      startQuiz();
      return;
    }
    // BACK button
    if (t.id === 'btn-back') {
      e.preventDefault();
      prevQuestion();
      return;
    }
    // NEXT / SUBMIT QUIZ button
    if (t.id === 'btn-next') {
      e.preventDefault();
      nextQuestion();
      return;
    }
    // SUBMIT EMAIL button
    if (t.id === 'btn-submit') {
      e.preventDefault();
      showResult();
      return;
    }
    // OPTION buttons (data-index set in renderQuestion)
    if (t.hasAttribute && t.hasAttribute('data-index')) {
      e.preventDefault();
      selectOption(parseInt(t.getAttribute('data-index')));
      return;
    }

    t = t.parentElement;
  }
});
